# ICM-42688-P 传感器驱动代码详细注解

## 文件概述
这是一个用于 ESP-IDF 的 ICM-42688-P 6轴IMU（加速度计+陀螺仪）传感器的 SPI 驱动程序。代码基于 Arduino HAL 模型，但适配到 ESP-IDF 的 SPI 接口。

## 主要功能模块

### 1. 头文件包含和宏定义

```c
// 包含必要的头文件
#include "icm42688p.h"        // 传感器驱动头文件
#include "logger.h"          // 日志系统
#include "driver/spi_master.h" // ESP-IDF SPI 主设备驱动
#include "freertos/FreeRTOS.h" // FreeRTOS 操作系统
#include "freertos/task.h"    // FreeRTOS 任务管理
#include "esp_timer.h"       // ESP32 高精度定时器

// 寄存器地址定义
#define ICM42688P_REG_WHO_AM_I   0x75   // 设备ID寄存器
#define ICM42688P_REG_PWR_MGMT0  0x4E   // 电源管理寄存器
#define ICM42688P_REG_ACCEL_X1   0x1F   // 加速度计X轴数据高字节

// SPI 配置
#define ICM42688P_SPI_HZ 10 * 1000 * 1000  // SPI 时钟频率：10MHz
#define ICM42688P_ACCEL_LSB_PER_G 2048.0f  // 加速度计灵敏度：2048 LSB/g
```

### 2. 全局变量

```c
static spi_device_handle_t s_icm_dev = NULL;  // SPI 设备句柄
static int16_t s_last_ax = 0, s_last_ay = 0, s_last_az = 0;  // 缓存的加速度数据
static int64_t s_last_read_us = -1;           // 上次读取时间（微秒）
static uint32_t s_last_interval_us = 0;       // 上次使用的读取间隔
```

### 3. SPI 总线初始化 (`icm_spi_init_bus`)

```c
static esp_err_t icm_spi_init_bus(void)
{
    spi_bus_config_t bus_cfg = {
        .mosi_io_num = ICM42688P_SPI_MOSI,  // MOSI 引脚（主出从入）
        .miso_io_num = ICM42688P_SPI_MISO,  // MISO 引脚（主入从出）
        .sclk_io_num = ICM42688P_SPI_SCK,   // 时钟引脚
        .quadwp_io_num = -1,               // 不使用 QSPI WP 引脚
        .quadhd_io_num = -1,               // 不使用 QSPI HD 引脚
        .max_transfer_sz = 16,             // 最大传输大小（字节）
    };
    
    // 初始化 SPI 总线，使用自动分配的 DMA 通道
    esp_err_t err = spi_bus_initialize(ICM42688P_SPI_HOST, &bus_cfg, SPI_DMA_CH_AUTO);
    
    // 如果总线已经初始化（ESP_ERR_INVALID_STATE），则认为是正常的
    if (err == ESP_ERR_INVALID_STATE) {
        return ESP_OK;
    }
    return err;
}
```

**关键点**：
- 使用 SPI 模式 0（CPOL=0, CPHA=0）
- 支持 DMA 传输以提高效率
- 处理总线重复初始化的特殊情况

### 4. SPI 设备添加 (`icm_spi_add_device`)

```c
static esp_err_t icm_spi_add_device(void)
{
    spi_device_interface_config_t dev_cfg = {
        .clock_speed_hz = ICM42688P_SPI_HZ,  // 10MHz 时钟
        .mode = 0,                           // SPI 模式 0
        .spics_io_num = ICM42688P_SPI_CS,     // 片选引脚
        .queue_size = 2,                     // 传输队列大小
        .flags = 0,                          // 全双工模式
    };
    return spi_bus_add_device(ICM42688P_SPI_HOST, &dev_cfg, &s_icm_dev);
}
```

**关键点**：
- 配置 SPI 设备参数
- 使用全双工模式（支持同时发送和接收）
- 创建 SPI 设备句柄供后续使用

### 5. 寄存器读写函数

#### 写寄存器 (`icm_write_reg`)

```c
static esp_err_t icm_write_reg(uint8_t reg, uint8_t val)
{
    // SPI 传输格式：第一个字节是寄存器地址（最高位为0表示写操作）
    // 第二个字节是要写入的值
    uint8_t tx[2] = { (uint8_t)(reg & 0x7F), val };
    
    spi_transaction_t t = {
        .length = 16,        // 传输16位（2字节）
        .tx_buffer = tx,     // 发送缓冲区
    };
    
    // 使用轮询方式传输（阻塞直到完成）
    return spi_device_polling_transmit(s_icm_dev, &t);
}
```

**关键点**：
- 寄存器地址的最高位为0表示写操作
- 使用轮询传输确保时序准确

#### 读寄存器 (`icm_read_reg`)

```c
static esp_err_t icm_read_reg(uint8_t reg, uint8_t *out)
{
    if (!out) return ESP_ERR_INVALID_ARG;
    
    // SPI 传输格式：第一个字节是寄存器地址（最高位为1表示读操作）
    // 第二个字节是虚拟数据（实际接收数据在第二个字节）
    uint8_t tx[2] = { (uint8_t)(reg | 0x80), 0 };
    uint8_t rx[2] = {0};
    
    spi_transaction_t t = {
        .length = 16,        // 传输16位
        .tx_buffer = tx,     // 发送缓冲区
        .rx_buffer = rx,     // 接收缓冲区
        .flags = 0,
    };
    
    esp_err_t err = spi_device_polling_transmit(s_icm_dev, &t);
    if (err == ESP_OK) {
        *out = rx[1];  // 第二个字节是读取的数据
    }
    return err;
}
```

**关键点**：
- 寄存器地址的最高位为1表示读操作
- 读取的数据在传输的第二个字节中

### 6. 加速度计原始数据读取 (`icm_read_accel_raw`)

```c
static esp_err_t icm_read_accel_raw(int16_t *ax, int16_t *ay, int16_t *az)
{
    if (!ax || !ay || !az) return ESP_ERR_INVALID_ARG;

    // 从 ACCEL_X1 寄存器开始连续读取7个字节
    // 第一个字节：寄存器地址 + 读标志
    // 后续6个字节：X高、X低、Y高、Y低、Z高、Z低
    uint8_t tx[7] = { (uint8_t)(ICM42688P_REG_ACCEL_X1 | 0x80), 0,0,0,0,0,0 };
    uint8_t rx[7] = {0};
    
    spi_transaction_t t = {
        .length = 7 * 8,     // 7字节 = 56位
        .tx_buffer = tx,
        .rx_buffer = rx,
        .flags = 0,
    };
    
    esp_err_t err = spi_device_polling_transmit(s_icm_dev, &t);
    if (err != ESP_OK) return err;

    // 将两个字节组合成16位有符号整数
    *ax = (int16_t)((rx[1] << 8) | rx[2]);  // X轴
    *ay = (int16_t)((rx[3] << 8) | rx[4]);  // Y轴
    *az = (int16_t)((rx[5] << 8) | rx[6]);  // Z轴
    return ESP_OK;
}
```

**关键点**：
- 使用突发读取（burst read）一次性读取所有加速度数据
- 数据格式：每个轴2字节，高字节在前
- 提高读取效率，减少SPI传输次数

### 7. 带速率限制的加速度读取 (`icm42688p_read_accel_raw_rate`)

```c
esp_err_t icm42688p_read_accel_raw_rate(int16_t *ax, int16_t *ay, int16_t *az, uint32_t interval_us)
{
    if (!ax || !ay || !az) return ESP_ERR_INVALID_ARG;

    int64_t now = esp_timer_get_time();  // 获取当前时间（微秒）
    
    // 判断是否需要重新读取：
    // 1. interval_us == 0：总是重新读取
    // 2. 有缓存数据且未超过时间间隔：使用缓存
    // 3. 否则：重新读取
    bool reuse = (interval_us > 0) &&
                 (s_last_read_us >= 0) &&
                 (now - s_last_read_us < (int64_t)interval_us);

    if (!reuse) {
        esp_err_t err = icm_read_accel_raw(&s_last_ax, &s_last_ay, &s_last_az);
        if (err != ESP_OK) return err;
        s_last_read_us = now;
        s_last_interval_us = interval_us;
    }

    *ax = s_last_ax;
    *ay = s_last_ay;
    *az = s_last_az;
    return ESP_OK;
}
```

**关键点**：
- 实现数据缓存机制，避免频繁读取
- 使用高精度定时器（esp_timer）进行时间管理
- 支持可配置的读取间隔，平衡性能和功耗

### 8. 传感器初始化 (`icm42688p_init`)

```c
esp_err_t icm42688p_init(void)
{
    // 1. 初始化SPI总线
    esp_err_t err = icm_spi_init_bus();
    if (err != ESP_OK && err != ESP_ERR_INVALID_STATE) {
        LOG_ERRORF("ICM SPI bus init failed: %d", err);
        return err;
    }

    // 2. 添加SPI设备
    err = icm_spi_add_device();
    if (err != ESP_OK) {
        LOG_ERRORF("ICM add device failed: %d", err);
        return err;
    }

    // 3. 等待传感器上电稳定
    vTaskDelay(pdMS_TO_TICKS(50));

    // 4. 读取设备ID验证通信
    uint8_t id = 0;
    err = icm_read_reg(ICM42688P_REG_WHO_AM_I, &id);
    if (err != ESP_OK) {
        LOG_ERRORF("ICM read WHO_AM_I failed: %d", err);
        return err;
    }
    if (id == 0x00 || id == 0xFF) {
        LOG_ERROR("ICM invalid WHO_AM_I");
        return ESP_FAIL;
    }

    // 5. 配置电源管理（启用加速度计和陀螺仪）
    err = icm_write_reg(ICM42688P_REG_PWR_MGMT0, 0x0F);
    if (err != ESP_OK) {
        LOG_ERRORF("ICM write PWR_MGMT0 failed: %d", err);
        return err;
    }
    
    // 6. 等待配置生效
    vTaskDelay(pdMS_TO_TICKS(10));
    return ESP_OK;
}
```

**关键点**：
- 完整的初始化流程
- 设备ID验证确保硬件连接正确
- 电源管理配置：0x0F = 启用加速度计和陀螺仪
- 适当的延时确保配置生效

#### 读取加速度计数据（转换为g单位）(`icm42688p_read_accel`)
```c
esp_err_t icm42688p_read_accel(icm42688p_sample_t *out_sample)
{
    if (!out_sample) return ESP_ERR_INVALID_ARG;
    
    // 1. 读取原始数据
    int16_t ax = 0, ay = 0, az = 0;
    esp_err_t err = icm_read_accel_raw(&ax, &ay, &az);
    if (err != ESP_OK) return err;

    // 2. 转换为g单位（重力加速度）
    // 公式：g值 = 原始值 / 灵敏度(2048 LSB/g)
    out_sample->x_g = (float)ax / ICM42688P_ACCEL_LSB_PER_G;
    out_sample->y_g = (float)ay / ICM42688P_ACCEL_LSB_PER_G;
    out_sample->z_g = (float)az / ICM42688P_ACCEL_LSB_PER_G;
    return ESP_OK;
}
```

- 避免频繁读取传感器

## 使用示例

```c
// 初始化传感器
if (icm42688p_init() != ESP_OK) {
    printf("传感器初始化失败\n");
    return;
}

// 读取设备ID
uint8_t id;
icm42688p_who_am_i(&id);
printf("设备ID: 0x%02X\n", id);

// 读取加速度数据（带缓存，最大100Hz）
int16_t ax, ay, az;
while (1) {
    if (icm42688p_read_accel_raw_rate(&ax, &ay, &az, 10000) == ESP_OK) {
        printf("加速度: X=%d, Y=%d, Z=%d\n", ax, ay, az);
    }
    vTaskDelay(pdMS_TO_TICKS(10));
}
```

## 注意事项

1. **引脚配置**：需要在 `icm42688p.h` 中正确配置SPI引脚
2. **电源管理**：当前代码只启用了加速度计，如需陀螺仪需要额外配置
3. **数据范围**：默认量程为±16g（2048 LSB/g）
4. **SPI共享**：如果多个设备共享SPI总线，需要注意片选信号管理

这个驱动程序为ICM-42688-P传感器提供了稳定、高效的接口，适合嵌入式振动监测等应用场景。
